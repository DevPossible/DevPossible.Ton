API Documentation

{(apiEndpoint)
    path = "/api/users/{id}",
    method = "GET",
    description = """
    Retrieve a user by their ID.
    Returns the user object with profile information.
    """,

    parameters = [
        {(param)
            name = "id",
            type = "integer",
            required = true,
            description = "User ID"
        }
    ],

    response = '''
    {
        "id": 123,
        "name": "John Doe",
        "email": "john@example.com",
        "createdAt": "2024-01-15T10:30:00Z"
    }
    ''',

    examples = {
        curl = "curl -X GET https://api.example.com/api/users/123",
        javascript = """
        const response = await fetch('/api/users/123');
        const user = await response.json();
        """
    }
}

Docker Compose Configuration

{(dockerCompose)
    version = "3.8",

    services = {
        web = {(service)
            image = "nginx:alpine",
            ports = ["80:80", "443:443"],
            volumes = [
                "./nginx.conf:/etc/nginx/nginx.conf",
                "./ssl:/etc/ssl"
            ],
            dependsOn = ["app"]
        },

        app = {(service)
            build = ".",
            environment = {
                NODE_ENV = "production",
                DATABASE_URL = "postgresql://user:pass@db:5432/myapp"
            },
            ports = ["3000:3000"]
        },

        db = {(service)
            image = "postgres:15",
            environment = {
                POSTGRES_DB = "myapp",
                POSTGRES_USER = "user",
                POSTGRES_PASSWORD = "password"
            },
            volumes = ["postgres_data:/var/lib/postgresql/data"]
        }
    },

    volumes = {
        postgres_data = {}
    }
}
